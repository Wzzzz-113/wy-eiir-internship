# 1. 生成 SSH 密钥（如果还没有）
ssh-keygen -t ed25519 -C "your_email@example.com"
# 直接按回车使用默认设置

# 2. 查看并复制公钥
cat ~/.ssh/id_ed25519.pub

# 3. 添加到 GitHub：Settings → SSH and GPG keys → New SSH key

# 4. 修改远程地址为 SSH
git remote set-url origin git@github.com:Wzzzz-113/YOLO.git

# 5. 之后推送都不需要输入密码
git push origin main
